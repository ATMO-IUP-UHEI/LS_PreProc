netcdf ATM_example {
dimensions:
	line = 3 ;
	sample = 4 ;
	level = 60 ;
variables:
	int64 datetime ;
		datetime:standard_name = "date and time of measurement" ;
		datetime:units = "days since 2019-08-06 14:52:33" ;
		datetime:calendar = "proleptic_gregorian" ;
	double latitude(line, sample) ;
		latitude:standard_name = "latitude" ;
		latitude:units = "degrees north" ;
	double longitude(line, sample) ;
		longitude:standard_name = "longitude" ;
		longitude:units = "degrees east" ;
	double surface_elevation(line, sample) ;
		surface_elevation:standard_name = "surface elevation" ;
		surface_elevation:units = "m" ;
	double pressure(level, line, sample) ;
		pressure:standard_name = "pressure" ;
		pressure:units = "kg m-1 s-2" ;
	double temperature(level, line, sample) ;
		temperature:standard_name = "temperature" ;
		temperature:units = "K" ;
	double geometric_altitude(level, line, sample) ;
		geometric_altitude:standard_name = "geometric altitude" ;
		geometric_altitude:units = "m" ;
	double h2o(level, line, sample) ;
		h2o:standard_name = "H2O mole fraction" ;
		h2o:units = "mol mol-1" ;
	double co2(level, line, sample) ;
		co2:standard_name = "CO2 mole fraction" ;
		co2:units = "mol mol-1" ;
	double ch4(level, line, sample) ;
		ch4:standard_name = "CH4 mole fraction" ;
		ch4:units = "mol mol-1" ;
}
